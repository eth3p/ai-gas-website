document.querySelectorAll('a[href*="affiliate"]').forEach(link => {

        gtag('event', 'affiliate_click', {
            'affiliate_program': this.dataset.program,
            'link_text': this.textContent,
            'page_location': window.location.pathname
        });
    });
});
</script>
